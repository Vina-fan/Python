class ClusterUnit:
    def __init__(self):
        self.node_list = []  # 每一个类对应的节点列表
        self.node_num = 0  # 该类下的节点数
        self.centroid = None  # 类中心质心
 
    def add_node(self, node, 
                 node_vec):
        # 根据添加的点更新类中心，node_vec:添加的点对应的特征向量，node:节点
        self.node_list.append(node)
        try:
            self.centroid = (self.node_num * self.centroid + node_vec) / (self.node_num + 1)  # 更新类中心
        except TypeError:
            self.centroid = np.array(node_vec) * 1  # 初始化类中心
        self.node_num += 1  # 节点数加1
 
    def remove_node(self, node):
        # 移除类中的节点
        try:
            self.node_list.remove(node)
            self.node_num -= 1
        except ValueError:
            raise ValueError("%类中不存在该节点" % node)  # 该簇本身就不存在该节点，移除失败
 
    def move_node(self, node, another_cluster):
        # 将本类中的节点移至另一个簇
        self.remove_node(node=node)
        another_cluster.add_node(node=node)

def euclidian_distance(vec_a, vec_b):
    # 计算向量a与向量b的距离，此处采用欧氏距离
    diff = vec_a - vec_b
    return sqrt(np.dot(diff, diff))  # dot计算矩阵内积
 
class Leader:
    def __init__(self, t, vector_list):
        self.threshold = t  # 指定类与样本之间的最大距离也就是阈值
        self.vectors = np.array(vector_list)
        self.cluster_list = []  # 聚类后类的列表
        t1 = time.time()
        self.clustering()
        t2 = time.time()
        self.cluster_num = len(self.cluster_list)  # 聚类完成后类的个数
        self.spend_time = t2 - t1  # 聚类占用时间
 
    def clustering(self):
        self.cluster_list.append(ClusterUnit())  # 初始新建一个类
        self.cluster_list[0].add_node(0, self.vectors[0])  # 将读入的第一个节点归于该簇
        for index in range(len(self.vectors))[1:]:
            min_distance = euclidian_distance(vec_a=self.vectors[0],vec_b=self.cluster_list[0].centroid)  # 与类的中心的最小距离
            min_cluster_index = 0  # 最小距离的类的索引
            for cluster_index, cluster in enumerate(self.cluster_list[1:]):# enumerate会将数组或列表组成一个索引序列，寻找距离最小的类，记录下距离和对应的类的索引
                distance = euclidian_distance(vec_a=self.vectors[index],
                                              vec_b=cluster.centroid)
                if distance < min_distance:
                    min_distance = distance
                    min_cluster_index = cluster_index + 1
            if min_distance < self.threshold:  # 最小距离小于阈值，则归于该类
                self.cluster_list[min_cluster_index].add_node(index, self.vectors[index])
            else:  # 否则该样本自成一类
                new_cluster = ClusterUnit()
                new_cluster.add_node(index, self.vectors[index])
                self.cluster_list.append(new_cluster)
                del new_cluster
 
    def print_result(self, label_dict=None):# label_dict:节点对应的标签字典
        print ("Leader聚类结果：")
        for index, cluster in enumerate(self.cluster_list):
            print ("类列表:%s" % index)  # 类的序号
            print (cluster.node_list)   # 该类的节点列表
            if label_dict is not None:
                print (" ".join([label_dict[n] for n in cluster.node_list]) ) # 若有提供标签字典，则输出该类的标签
            print ("节点数量: %s" % cluster.node_num)
            print ("-------------")
        print ("本案例中节点数共有 %s" % len(self.vectors))
        print ("本案例中聚类数共有%s" % self.cluster_num)
        print ("花费时间 %.9fs" % (self.spend_time / 1000))

K = 12
n = 100#缓冲区的数量
end = 100
time_start=time.time()
features_all = temperature#用于记录坐标
buffset = features_all[:end]#初始化缓冲区
clf = KMeans(n_clusters=K)
clf.fit(buffset)  #一开始对缓冲区聚类
centers = clf.cluster_centers_ # 聚类中心
labels = clf.labels_   # 聚类标签
for i in range(K):
    buffset[np.where(labels==i)]=centers[i:(i+1)]

#对后面进来的缓冲区数据进行聚类    
for i in range(int(len(temperature)/n)):#循环
    if ((end+n) <= len(temperature)):
        buffset_new = temperature[end:(end+n)]
    else:
        buffset_new = temperature[end:]
    #在缓冲区的基础上更新数据集，对于k个类中心，赋予权重
    buffset = np.concatenate((buffset,buffset_new),0)
    clf = KMeans(n_clusters=K)
    clf.fit(buffset)  #对更新后的数据集进行聚类
    centers = clf.cluster_centers_ # 更新聚类中心
    labels = clf.labels_   # 更新聚类标签
    for i in range(K):
        buffset[np.where(labels==i)]=centers[i:(i+1)]
    end+=n
    
time_end=time.time()
print('耗时',time_end-time_start)
y_hat = labels
